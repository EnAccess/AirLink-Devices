700void ble_evt_handler(ble_evt_t const * p_ble_evt, void * p_context)
701{
702 uint32_t err_code = NRF_SUCCESS;
703
704 switch (p_ble_evt->header.evt_id)
705 {
706 case BLE_GAP_EVT_DISCONNECTED:
707 // LED indication will be changed when advertising starts.
708 break;
709
710 case BLE_GAP_EVT_CONNECTED:
711 err_code = bsp_indication_set(BSP_INDICATE_CONNECTED);
712 APP_ERROR_CHECK(err_code);
713 m_conn_handle = p_ble_evt->evt.gap_evt.conn_handle;
714 err_code = nrf_ble_qwr_conn_handle_assign(&m_qwr, m_conn_handle);
715 APP_ERROR_CHECK(err_code);
716 break;
717
718 case BLE_GAP_EVT_PHY_UPDATE_REQUEST:
719 {
720 NRF_LOG_DEBUG("PHY update request.");
721 ble_gap_phys_t const phys =
722 {
723 .rx_phys = BLE_GAP_PHY_AUTO,
724 .tx_phys = BLE_GAP_PHY_AUTO,
725 };
726 err_code = sd_ble_gap_phy_update(p_ble_evt->evt.gap_evt.conn_handle, &phys);
727 APP_ERROR_CHECK(err_code);
728 break;
729 }
730
731 case BLE_GATTC_EVT_TIMEOUT:
732 // Disconnect on GATT Client timeout event.
733 NRF_LOG_DEBUG("GATT Client Timeout.");
734 err_code = sd_ble_gap_disconnect(p_ble_evt->evt.gattc_evt.conn_handle,
735 BLE_HCI_REMOTE_USER_TERMINATED_CONNECTION);
736 APP_ERROR_CHECK(err_code);
737 break;
738
739 case BLE_GATTS_EVT_TIMEOUT:
740 // Disconnect on GATT Server timeout event.
741 NRF_LOG_DEBUG("GATT Server Timeout.");
742 err_code = sd_ble_gap_disconnect(p_ble_evt->evt.gatts_evt.conn_handle,
743 BLE_HCI_REMOTE_USER_TERMINATED_CONNECTION);
744 APP_ERROR_CHECK(err_code);
745 break;
746
747 default:
748 // No implementation needed.
749 break;
750 }
751}
