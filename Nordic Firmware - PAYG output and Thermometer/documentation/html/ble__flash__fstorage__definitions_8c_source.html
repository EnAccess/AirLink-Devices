108 static void fstorage_evt_handler(nrf_fstorage_evt_t * p_evt)
109 {
110 if (p_evt->result != NRF_SUCCESS)
111 {
112 NRF_LOG_INFO("--> Event received: ERROR while executing an fstorage operation.");
113 return;
114 }
115
116 switch (p_evt->id)
117 {
118 case NRF_FSTORAGE_EVT_WRITE_RESULT:
119 {
120 NRF_LOG_INFO("--> Event received: wrote %d bytes at address 0x%x.",
121 p_evt->len, p_evt->addr);
122 } break;
123
124 case NRF_FSTORAGE_EVT_ERASE_RESULT:
125 {
126 NRF_LOG_INFO("--> Event received: erased %d page from address 0x%x.",
127 p_evt->len, p_evt->addr);
128 } break;
129
130 default:
131 break;
132 }
133 }

256static void ssl_low_fstorage_write(uint8_t page_number){
257 ret_code_t rc;
258 const uint32_t page_start_address[] = {page1_start_address, page2_start_address, page3_start_address};
259 //uint64_t page_end_address;
260 char flag[4];
261 uint32_t start_address=0;
262 const uint16_t ssl_block_size[] = {100, 48, 4096};
263 const uint8_t ssl_block_per_page[] = {40, 85, 1};
264 uint16_t index;
265
266 page_number -= 1;
267 // If page is not 1 or 2 do nothing
268 if(page_number > 1){
269 return;
270 }
271 if(CYCLIC_SAVE_METHOD_ENABLED){
272 // Find block to save flag is supposed to be 0x27
273 for (index=0;index<ssl_block_per_page[page_number];index++){
274 rc = nrf_fstorage_read(&fstorage, page_start_address[page_number]+(ssl_block_size[page_number]*index), flag, 4);
275 if(flag[0]!='\x27'){
276 start_address = page_start_address[page_number]+(ssl_block_size[page_number]*index);
277 break;
278 }
279 }
280 }
281 if(start_address == 0){ // page is full erase and start from the start of page
282 rc = nrf_fstorage_erase(&fstorage, page_start_address[page_number], 1, NULL);
283 APP_ERROR_CHECK(rc);
284 wait_for_flash_ready(&fstorage);
285 NRF_LOG_INFO("Erase Done.");
286 start_address = page_start_address[page_number];
287 }
288 if(page_number == 0){
289 // Prepare string to be written to flash
290 char *ptr;
291 ptr = (char*)malloc(ssl_block_size[page_number]*sizeof(char));
292
293 // write flag (1 of 4 Bytes)
294 uint16_t x=0;
295 sprintf(ptr+x,"%c",'\x27');
296 x+=4;
297 // write owner's name (16 of 16 Bytes)
298 sprintf(ptr+x,"%s",user_name);
299 x+=16;
300 // write owner's phone (16 of 16 Bytes)
301 sprintf(ptr+x,"%s",user_phone);
302 x+=16;
303 // write Device ID (4 of 8 Bytes)
304 //sprintf(ptr+x,"%s",device_id);
305 ptr[x] = ((device_id) & 0xff);
306 ptr[x+1] = ((device_id >> 8) & 0xff);
307 ptr[x+2] = ((device_id >> 16) & 0xff);
308 ptr[x+3] = ((device_id >> 24) & 0xff);
309 x+=8;
310 // write device asset status (1 0f 4 Bytes)
311 //sprintf(ptr+x,"%c",dev_asset_status[0]);
312 ptr[x] = dev_asset_status;
313 x+=4;
314 // write resource_model_type (1 of 4 Bytes) // Can be replaced
315 //sprintf(ptr+x,"%c",resource_model_type[0]);
316 ptr[x] = resource_model_type;
317 x+=4;
318 // write manufacturer_id (3 of 4 Bytes) // Can be replaced
319 sprintf(ptr+x,"%s",manufacturer_id);
320 x+=4;
321 // write advt_data_format (2 of 4 Bytes) // Can be replaced
322 //sprintf(ptr+x,"%s",advt_data_format);
323 ptr[x+2] = ((advt_data_format >> 8) & 0xff);
324 ptr[x+3] = (advt_data_format & 0xff);
325 x+=4;
326 // write payG_unit (1 of 4 Bytes) // Can be replaced
327 sprintf(ptr+x,"%c",payG_unit);
328
329 rc = nrf_fstorage_write(&fstorage, start_address, ptr, ssl_block_size[page_number], NULL);
330 APP_ERROR_CHECK(rc);
331 wait_for_flash_ready(&fstorage);
332 free(ptr);
333 flash_upload_flag &= 0xfe; // reset the flag
334 }
335 else if(page_number == 1){
336 uint16_t x=0;
337 uint32_t *ptr;
338 ptr = (uint32_t *)malloc(ssl_block_size[page_number]*sizeof(uint32_t));
339
340 // write flag (1 of 4 Bytes)
341 ptr[x]=0x27;
342 x+=1;
343 // write last_payG_update (4 of 8 Bytes)
344 ptr[x]=last_payG_update;
345 x+=2;
346 // write system_uptime (4 of 4 Bytes)
347 ptr[x]=system_uptime;
348 x+=1;
349 // write payG_credit_remaining (2 of 4 Bytes)
350 ptr[x]=payG_credit_remaining;
351 x+=1;
352 // write error_status (1 of 4 Bytes)
353 ptr[x]=error_status;
354 x+=1;
355 // write batt_voltage (2 of 4 Bytes)
356 ptr[x]=batt_voltage;
357 x+=1;
358 // write batt_current (2 of 4 Bytes)
359 ptr[x]=batt_current;
360 x+=1;
361
362 rc = nrf_fstorage_write(&fstorage, start_address, ptr, ssl_block_size[page_number], NULL);
363 APP_ERROR_CHECK(rc);
364 wait_for_flash_ready(&fstorage);
365 free(ptr);
366 flash_upload_flag &= 0XFD;
367 }
368
369 NRF_LOG_INFO("Write Done.");
370
