67 else // if there is variable overloading condition then zero
68 system_uptime_hrs_diff = 0;
69
70 if(system_uptime_hrs_diff){ // Check if the difference in time is more than an hour

72 payG_credit_remaining -= system_uptime_hrs_diff; // Decrement credit remaining
73 else // if there is variable overloading condition then zero remaing credit
74 payG_credit_remaining=0;
